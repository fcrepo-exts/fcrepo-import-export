018package org.fcrepo.importexport.common;
019
020import static org.apache.commons.codec.binary.Hex.encodeHex;
021
022import java.io.BufferedWriter;
023import java.io.File;
024import java.io.FileOutputStream;
025import java.io.FileWriter;
026import java.io.IOException;
027import java.io.OutputStream;
028import java.io.PrintWriter;
029import java.security.MessageDigest;
030import java.security.NoSuchAlgorithmException;
031import java.util.HashMap;
032import java.util.Iterator;
033import java.util.Map;
034import java.util.Set;
035
036/**
037 * Utility to write BagIt bags.
038 *
039 * @author escowles
040 * @since 2016-12-15
041 */
042public class BagWriter {
043
044    private File bagDir;
045    private File dataDir;
046    private Set<String> algorithms;
047
048    private Map<String, Map<File, String>> payloadRegistry;
049    private Map<String, Map<File, String>> tagFileRegistry;
050    private Map<String, Map<String, String>> tagRegistry;
051
052    /**
053     * Version of the BagIt specification implemented
054     */
055    public static String BAGIT_VERSION = "0.97";
056
057    /**
058     * Create a new, empty Bag
059     * @param bagDir The base directory for the Bag (will be created if it doesn't exist)
060     * @param algorithms Set of digest algorithms to use for manifests (e.g., "md5", "sha1", or "sha256")
061     */
062    public BagWriter(final File bagDir, final Set<String> algorithms) {
063        this.bagDir = bagDir;
064        this.dataDir = new File(bagDir, "data");
065        if (!dataDir.exists()) {
066            dataDir.mkdirs();
067        }
068
069        this.algorithms = algorithms;
070        payloadRegistry = new HashMap<>();
071        tagFileRegistry = new HashMap<>();
072        tagRegistry = new HashMap<>();
073
074        final Map<String, String> bagitValues = new HashMap<>();
075        bagitValues.put("BagIt-Version", BAGIT_VERSION);
076        bagitValues.put("Tag-File-Character-Encoding", "UTF-8");
077        tagRegistry.put("bagit.txt", bagitValues);
078    }
079
080    /**
081     * Get the Bag's root directory
082     * @return File object for the directory
083     */
084    public File getRootDir() {
085        return bagDir;
086    }
087
088    /**
089     * Register checksums of payload (data) files
090     * @param algorithm Checksum digest algorithm name (e.g., "SHA-1")
091     * @param filemap Map of Files to checksum values
092     */
093    public void registerChecksums(final String algorithm, final Map<File, String> filemap) {
094        if (!algorithms.contains(algorithm)) {
095            throw new RuntimeException("Invalid algorithm: " + algorithm);
096        }
097        payloadRegistry.put(algorithm, filemap);
098    }
099
100    /**
101     * Add tags (metadata) to the Bag
102     * @param key Filename of the tag file (e.g., "bag-info.txt")
103     * @param values Map containing field/value pairs
104     */
105    public void addTags(final String key, final Map<String, String> values) {
106        tagRegistry.put(key, values);
107    }
108
109    /**
110     * Get the current tag (metadata) of the Bag
111     * @param key Filename of the tag file (e.g., "bag-info.txt")
112     * @return Map of field/value pairs
113     */
114    public Map<String, String> getTags(final String key) {
115        return tagRegistry.get(key);
116    }
117
118    /**
119     * Write metadata and finalize Bag
120     * @throws IOException when an I/O error occurs
121     * @throws NoSuchAlgorithmException when an unsupported algorithm is used
122     */
123    public void write() throws IOException, NoSuchAlgorithmException {
124        writeManifests("manifest", payloadRegistry);
125        for (final Iterator<String> it = tagRegistry.keySet().iterator(); it.hasNext(); ) {
126            writeTagFile(it.next());
127        }
128        writeManifests("tagmanifest", tagFileRegistry);
129    }
130
131    private void writeManifests(final String prefix, final Map<String, Map<File, String>> registry)
132            throws IOException {
133        for (final Iterator<String> it = algorithms.iterator(); it.hasNext(); ) {
134            final String algorithm = it.next();
135            final Map<File, String> filemap = registry.get(algorithm);
136            if (filemap != null) {
137                final File f = new File(bagDir, prefix + "-" + algorithm + ".txt");
138                try (PrintWriter out = new PrintWriter(new BufferedWriter(new FileWriter(f)))) {
139                    for (final Iterator<File> files = filemap.keySet().iterator(); files.hasNext(); ) {
140                        final File payload = files.next();
141                        out.println(filemap.get(payload) + "  " + bagDir.toPath().relativize(payload.toPath()));
142                    }
143                }
144            }
145        }
146    }
147
148    private void writeTagFile(final String key) throws IOException, NoSuchAlgorithmException {
149        final Map<String, String> values = tagRegistry.get(key);
150        final Map<String, String> checksums = new HashMap<>();
151        if (values != null) {
152            final File f = new File(bagDir, key);
153
154            MessageDigest md5 = null;
155            MessageDigest sha1 = null;
156            MessageDigest sha256 = null;
157            if (algorithms.contains("md5")) {
158                md5 = MessageDigest.getInstance("MD5");
159            }
160            if (algorithms.contains("sha1")) {
161                sha1 = MessageDigest.getInstance("SHA-1");
162            }
163            if (algorithms.contains("sha256")) {
164                sha256 = MessageDigest.getInstance("SHA-256");
165            }
166
167            try (OutputStream out = new FileOutputStream(f)) {
168                for (final Iterator<String> it = values.keySet().iterator(); it.hasNext(); ) {
169                    final String field = it.next();
170                    final byte[] bytes = (field + " : " + values.get(field) + "\n").getBytes();
171                    out.write(bytes);
172
173                    if (md5 != null) {
174                        md5.update(bytes);
175                    }
176                    if (sha1 != null) {
177                        sha1.update(bytes);
178                    }
179                    if (sha256 != null) {
180                        sha256.update(bytes);
181                    }
182                }
183            }
184
185            addTagChecksum("md5", f, md5);
186            addTagChecksum("sha1", f, sha1);
187            addTagChecksum("sha256", f, sha256);
188        }
189    }
190
191    private void addTagChecksum(final String algorithm, final File f, final MessageDigest digest) {
192        if (digest != null) {
193            Map<File, String> m = tagFileRegistry.get(algorithm);
194            if (m == null) {
195                m = new HashMap<>();
196                tagFileRegistry.put(algorithm, m);
197            }
198            m.put(f, new String(encodeHex(digest.digest())));
199        }
200    }
201}
