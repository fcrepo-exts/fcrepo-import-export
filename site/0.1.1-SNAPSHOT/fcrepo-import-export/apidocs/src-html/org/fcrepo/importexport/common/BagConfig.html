019package org.fcrepo.importexport.common;
020
021import java.io.File;
022import java.io.FileNotFoundException;
023import java.io.FileReader;
024import java.io.IOException;
025import java.util.Collections;
026import java.util.Map;
027import java.util.Set;
028
029import com.esotericsoftware.yamlbeans.YamlReader;
030
031/**
032 * A convenience class for parsing and storing bagit-config.yml information. The bagit-config.yml represents
033 * user-defined properties to be included in the bag-info.txt.
034 *
035 * @author dbernstein
036 * @since Dec 14, 2016
037 */
038public class BagConfig {
039
040    public enum AccessTypes {
041        RESTRICTED, INSTITUTION, CONSORTIA;
042    }
043
044    private static final String BAG_INFO_KEY = "bag-info.txt";
045
046    private static final String APTRUST_INFO_KEY = "aptrust-info.txt";
047
048    public static final String SOURCE_ORGANIZATION_KEY = "Source-Organization";
049
050    public static final String ORGANIZATION_ADDRESS_KEY = "Organization-Address";
051
052    public static final String CONTACT_NAME_KEY = "Contact-Name";
053
054    public static final String CONTACT_PHONE_KEY = "Contact-Phone";
055
056    public static final String CONTACT_EMAIL_KEY = "Contact-Email";
057
058    public static final String EXTERNAL_DESCRIPTION_KEY = "External-Description";
059
060    public static final String EXTERNAL_IDENTIFIER_KEY = "External-Identifier";
061
062    public static final String INTERNAL_SENDER_DESCRIPTION_KEY = "Internal-Sender-Description";
063
064    public static final String INTERNAL_SENDER_IDENTIFIER_KEY = "Internal-Sender-Identifier";
065
066    public static final String BAG_GROUP_IDENTIFIER = "Bag-Group-Identifier";
067
068    public static final String TITLE_KEY = "Title";
069
070    public static final String ACCESS_KEY = "Access";
071
072    private Map<String, Map<String, String>> map;
073
074    /**
075     * Default constructor
076     *
077     * @param bagConfigFile a bagit config yaml file (see src/test/resources/bagit-config.yml)
078     */
079    @SuppressWarnings("unchecked")
080    public BagConfig(final File bagConfigFile) {
081        final String bagConfigFilePath = bagConfigFile.getAbsolutePath();
082
083        YamlReader reader = null;
084        try {
085            reader = new YamlReader(new FileReader(bagConfigFile));
086            map = (Map<String, Map<String, String>>) reader.read();
087            if (getBagInfo() == null) {
088                throw new RuntimeException("The " + BAG_INFO_KEY + " key is not present in the " + bagConfigFilePath);
089            }
090
091
092        } catch (FileNotFoundException e) {
093            throw new RuntimeException("The specified bag config file does not exist: " + bagConfigFile
094                    .getAbsolutePath());
095        } catch (Exception e) {
096            throw new RuntimeException("The specified bag config file could not be parsed: " + e.getMessage(), e);
097        } finally {
098            if (reader != null) {
099                try {
100                    reader.close();
101                } catch (IOException e) {
102                }
103            }
104        }
105    }
106
107    /**
108     * Returns an immutable map of bag info properties.
109     *
110     * @return a map of bag info properties
111     */
112    public Map<String, String> getBagInfo() {
113        return Collections.unmodifiableMap(this.map.get(BAG_INFO_KEY));
114    }
115
116    /**
117     * Returns an immutable map of aptrust info properties.
118     *
119     * @return a map of aptrust info properties
120     */
121    public Map<String, String> getAPTrustInfo() {
122        return Collections.unmodifiableMap(this.map.get(APTRUST_INFO_KEY));
123    }
124
125    /**
126     * Returns all the tag files from the config
127     *
128     * @return set of tag filenames
129     */
130    public Set<String> getTagFiles() {
131        return map.keySet();
132    }
133
134    /**
135     * Check if a tag file is listed in bag config
136     *
137     * @param tagFile the tag filename
138     * @return true if it is list, false if not
139     */
140    public boolean hasTagFile(final String tagFile) {
141        return map.containsKey(tagFile);
142    }
143
144    /**
145     * Returns an immutable map of custom tags for a tag file
146     *
147     * @param tagFile name of the tag file to get fields for
148     * @return a map of filenames to key-value property maps
149     */
150    public Map<String, String> getFieldsForTagFile(final String tagFile) {
151        return map.get(tagFile);
152    }
153}
