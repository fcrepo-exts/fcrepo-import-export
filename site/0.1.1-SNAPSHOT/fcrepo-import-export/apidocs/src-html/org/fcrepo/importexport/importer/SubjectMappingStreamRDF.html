018package org.fcrepo.importexport.importer;
019
020import static org.apache.jena.graph.Factory.createDefaultGraph;
021import static org.apache.jena.rdf.model.ModelFactory.createModelForGraph;
022import static org.apache.jena.graph.NodeFactory.createURI;
023import static org.slf4j.LoggerFactory.getLogger;
024
025import java.net.URI;
026
027import org.apache.jena.rdf.model.Model;
028import org.slf4j.Logger;
029
030import org.apache.jena.atlas.lib.Sink;
031import org.apache.jena.graph.Graph;
032import org.apache.jena.graph.Node;
033import org.apache.jena.graph.Triple;
034import org.apache.jena.riot.lang.SinkTriplesToGraph;
035import org.apache.jena.riot.system.StreamRDFBase;
036
037/**
038 * StreamRDF implementation that maps URIs to a specified base URI.
039 *
040 * @author escowles
041 * @since 2016-09-08
042 */
043public class SubjectMappingStreamRDF extends StreamRDFBase {
044    private static final Logger logger = getLogger(SubjectMappingStreamRDF.class);
045
046    private final String sourceURI;
047    private final String baseURI;
048    private final Graph graph;
049    private final Sink<Triple> sink;
050
051    /**
052     * Create a subject-mapping RDF stream
053     * @param sourceURI the source URI to map triples from
054     * @param baseURI the base URI to map triples to
055     */
056    public SubjectMappingStreamRDF(final URI sourceURI, final URI baseURI) {
057        this.sourceURI = sourceURI.toString();
058        this.baseURI = baseURI.toString();
059        this.graph = createDefaultGraph();
060        this.sink = new SinkTriplesToGraph(true, graph);
061    }
062
063    @Override
064    public void triple(final Triple t) {
065        sink.send(Triple.create(rebase(t.getSubject()), t.getPredicate(), rebase(t.getObject())));
066    }
067
068    private Node rebase(final Node node) {
069        if (node.isURI() && node.getURI().startsWith(sourceURI)) {
070            return createURI(node.getURI().replaceFirst(sourceURI, baseURI));
071        }
072
073        return node;
074    }
075
076    @Override
077    public void finish() {
078        sink.flush();
079        sink.close();
080    }
081
082    /**
083     * Get the mapped triples as a model
084     * @return A model representing the triples sent to this stream
085     */
086    public Model getModel() {
087        return createModelForGraph(graph);
088    }
089}
