96      /**
97       * Constructor that takes the Import/Export configuration
98       *
99       * @param config for export
100      * @param clientBuilder for retrieving resources from Fedora
101      */
102     public Exporter(final Config config, final FcrepoClient.FcrepoClientBuilder clientBuilder) {
103         this.config = config;
104         this.clientBuilder = clientBuilder;
105         this.binaryURI = URI.create(NON_RDF_SOURCE.getURI());
106         this.containerURI = URI.create(CONTAINER.getURI());
107         this.exportLogger = config.getAuditLog();
108         this.dateFormat = new SimpleDateFormat("yyyy-MM-dd");
109 
110         if (config.getBagProfile() != null) {
111             try {
112                 // parse profile
113                 final URL url = this.getClass().getResource("/profiles/" + config.getBagProfile() + ".json");
114                 final BagConfig bagConfig = loadBagConfig(config.getBagConfigPath());
115                 final InputStream in = (url == null) ? new FileInputStream(config.getBagProfile()) : url.openStream();
116                 final BagProfile bagProfile = new BagProfile(in);
117 
118                 // always do sha1, do md5/sha256 if the profile asks for it
119                 final HashSet<String> algorithms = new HashSet<>();
120                 this.sha1FileMap = new HashMap<>();
121                 this.sha1 = MessageDigest.getInstance("SHA-1");
122                 algorithms.add("sha1");
123                 if (bagProfile.getPayloadDigestAlgorithms().contains("md5")) {
124                     this.md5FileMap = new HashMap<>();
125                     this.md5 = MessageDigest.getInstance("MD5");
126                     algorithms.add("md5");
127                 }
128                 if (bagProfile.getPayloadDigestAlgorithms().contains("sha256")) {
129                     this.sha256FileMap = new HashMap<>();
130                     this.sha256 = MessageDigest.getInstance("SHA-256");
131                     algorithms.add("sha256");
132                 }
133 
134                 //enforce default metadata
135                 bagProfile.validateConfig(bagConfig);
136 
137                 // setup bag
138                 final File bagdir = config.getBaseDirectory().getParentFile();
139                 this.bag = new BagWriter(bagdir, algorithms);
140                 for (final String tagFile : bagConfig.getTagFiles()) {
141                     this.bag.addTags(tagFile, bagConfig.getFieldsForTagFile(tagFile));
142                 }
143 
144             } catch (NoSuchAlgorithmException e) {
145                 // never happens with known algorithm names
146             } catch (FileNotFoundException e) {
147                 throw new RuntimeException(String.format("Error loading bag config file: {}", e.getMessage()), e);
148             } catch (IOException e) {
149                 throw new RuntimeException(String.format("Error reading bag profile: {}", e.getMessage()), e);
150             }
151         }
152     }

154     /**
155      * Loads a bag config from path
156      * @param bagConfigPath The path to the bag config yaml.
157      * @return
158      */
159     private BagConfig loadBagConfig(final String bagConfigPath) {
160         if (bagConfigPath == null) {
161             throw new RuntimeException("The bag config path must not be null.");
162         }
163         final File bagConfigFile = new File(bagConfigPath);
164         return new BagConfig(bagConfigFile);
165     }

172     private FcrepoClient client() {
173         if (config.getUsername() != null) {
174             clientBuilder.credentials(config.getUsername(), config.getPassword());
175         }
176         return clientBuilder.build();

179     /**
180      * This method does the export
181      */
182     @Override
183     public void run() {
184         logger.info("Running exporter...");
185         export(config.getResource());
186         if (bag != null) {
187             try {
188                 logger.info("Finishing bag manifests...");
189                 final Map<String, String> bagMetadata = new HashMap<>();
190                 bagMetadata.putAll(bag.getTags("bag-info.txt"));
191                 bagMetadata.putAll(bagTechMetadata());
192                 bag.addTags("bag-info.txt", bagMetadata);
193                 bag.registerChecksums("sha1", sha1FileMap);
194                 if (sha256 != null) {
195                     bag.registerChecksums("sha256", sha256FileMap);
196                 }
197                 if (md5 != null) {
198                     bag.registerChecksums("md5", md5FileMap);
199                 }
200                 bag.write();
201             } catch (IOException e) {
202                 throw new RuntimeException("Error finishing Bag: " + e.toString());
203             } catch (Exception e) {
204                 e.printStackTrace();
205             }
206         }
207         exportLogger.info("Finished export... {} bytes/{} resources exported", successBytes.get(),
208                 successCount.get());
209     }
210 
211     private Map<String, String> bagTechMetadata() {
212         final Map<String, String> metadata = new HashMap<>();
213         metadata.put("Bag-Size", byteCountToDisplaySize(successBytes.longValue()));
214         metadata.put("Payload-Oxum", successBytes.toString() + "." + successCount.toString());
215         metadata.put("Bagging-Date", dateFormat.format(new Date()));
216         return metadata;
217     }

219     private void export(final URI uri) {
220         try (FcrepoResponse response = client().head(uri).disableRedirects().perform()) {
221             checkValidResponse(response, uri);
222             final List<URI> linkHeaders = response.getLinkHeaders("type");
223             if (linkHeaders.contains(binaryURI)) {
224                 final String contentType = response.getContentType();
225                 final boolean external = contentType != null && contentType.contains("message/external-body");
226                 exportBinary(uri, external);
227             } else if (linkHeaders.contains(containerURI)) {
228                 exportDescription(uri);
229             } else {
230                 logger.error("Resource is neither an LDP Container nor an LDP NonRDFSource: {}", uri);
231                 exportLogger.error("Resource is neither an LDP Container nor an LDP NonRDFSource: {}", uri);
232             }
233         } catch (FcrepoOperationFailedException ex) {
234             logger.warn("Error retrieving content: {}", ex.toString());
235             exportLogger.error(String.format("Error retrieving context of uri: {}, Message: {}", uri, ex.toString()),
236                 ex);
237         } catch (IOException ex) {
238             logger.warn("Error writing content: {}", ex.toString());
239             exportLogger.error(String.format("Error writing content from uri: {}, Message: {}", uri, ex.toString()),
240                 ex);
241         }
242     }

244     private void exportBinary(final URI uri, final boolean external)
245             throws FcrepoOperationFailedException, IOException {
246 
247         if (!config.isIncludeBinaries()) {
248             logger.info("Skipping {}", uri);
249             return;
250         }
251 
252         try (FcrepoResponse response = client().get(uri).disableRedirects().perform()) {
253             checkValidResponse(response, uri);
254 
255             final File file = external ?
256                                 TransferProcess.fileForExternalResources(uri, config.getBaseDirectory()) :
257                                     TransferProcess.fileForBinary(uri, config.getBaseDirectory());
258 
259             logger.info("Exporting binary: {}", uri);
260             writeResponse(response, file);
261             exportLogger.info("export {} to {}", uri, file.getAbsolutePath());
262             successCount.incrementAndGet();
263         }
264     }
265 
266     private void exportDescription(final URI uri) throws FcrepoOperationFailedException, IOException {
267         final File file = TransferProcess.fileForURI(uri, config.getBaseDirectory(),
268                 config.getRdfExtension());
269         if (file == null) {
270             logger.info("Skipping {}", uri);
271             return;
272         }
273 
274         try (FcrepoResponse response = client().get(uri).accept(config.getRdfLanguage()).perform()) {
275             checkValidResponse(response, uri);
276             logger.info("Exporting description: {}", uri);
277             writeResponse(response, file);
278             exportLogger.info("export {} to {}", uri, file.getAbsolutePath());
279             successCount.incrementAndGet();
280         } catch ( Exception ex ) {
281             ex.printStackTrace();
282             exportLogger.error(String.format("Error exporting description: {}, Cause: {}", uri, ex.getMessage()), ex);
283         }
284 
285         exportMembers(file);
286     }
287     private void exportMembers(final File file) {
288         try {
289             final Model model = createDefaultModel().read(new FileInputStream(file), null, config.getRdfLanguage());
290             for (final String p : config.getPredicates()) {
291                 for (final NodeIterator it = model.listObjectsOfProperty(createProperty(p)); it.hasNext();) {
292                     export(URI.create(it.nextNode().toString()));
293                 }
294             }
295         } catch (FileNotFoundException ex) {
296             logger.warn("Unable to parse file: {}", ex.toString());
297         }
298     }
299     void writeResponse(final FcrepoResponse response, final File file)
300             throws IOException, FcrepoOperationFailedException {
301         if (!file.getParentFile().exists()) {
302             file.getParentFile().mkdirs();
303         }
304         try (OutputStream out = new FileOutputStream(file)) {
305             copy(response.getBody(), out);
306             logger.info("Exported {} to {}", response.getUrl(), file.getAbsolutePath());
307 
308             if (md5FileMap != null) {
309                 md5FileMap.put(file, new String(encodeHex(md5.digest())));
310             }
311             if (sha1FileMap != null) {
312                 sha1FileMap.put(file, new String(encodeHex(sha1.digest())));
313             }
314             if (sha256FileMap != null) {
315                 sha256FileMap.put(file, new String(encodeHex(sha256.digest())));
316             }
317         }
318 
319         final List<URI> describedby = response.getLinkHeaders("describedby");
320         for (final Iterator<URI> it = describedby.iterator(); describedby != null && it.hasNext(); ) {
321             exportDescription(it.next());
322         }
323     }

325     /**
326      * Copy bytes and generate checksums
327      * @param in Source data
328      * @param out Data destination
329      * @throws IOException If an I/O error occurs
330      */
331     private void copy(final InputStream in, final OutputStream out) throws IOException {
332         if (md5 != null) {
333             md5.reset();
334         }
335         if (sha1 != null) {
336             sha1.reset();
337         }
338         if (sha256 != null) {
339             sha256.reset();
340         }
341 
342         int read = 0;
343         final byte[] buf = new byte[8192];
344         while ((read = in.read(buf)) != -1) {
345             if (md5 != null) {
346                 md5.update(buf, 0, read);
347             }
348             if (sha1 != null) {
349                 sha1.update(buf, 0, read);
350             }
351             if (sha256 != null) {
352                 sha256.update(buf, 0, read);
353             }
354             if (out != null) {
355                 out.write(buf, 0, read);
356                 successBytes.addAndGet(read);
357             }
358         }
359     }

361     /**
362      * Checks the response code and throws a RuntimeException with a helpful
363      * message (when possible) for non 2xx codes.
364      * @param response the response from a REST call to Fedora
365      * @param uri the URI against which the request was made
366      */
367     private void checkValidResponse(final FcrepoResponse response, final URI uri) {
368         switch (response.getStatusCode()) {
369             case 401:
370                 throw new AuthenticationRequiredRuntimeException();
371             case 403:
372                 throw new AuthorizationDeniedRuntimeException(config.getUsername(), uri);
373             case 404:
374                 throw new ResourceNotFoundRuntimeException(uri);
375             default:
376                 if (response.getStatusCode() < 200 || response.getStatusCode() > 307) {
377                     throw new RuntimeException("Export operation failed: unexpected status "
378                             + response.getStatusCode() + " for " + uri);
379                 }
380         }
